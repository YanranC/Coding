import java.util.*;
// @lc code=start
class Solution {
    public int removeDuplicates(int[] nums) {
        int slow=0; fast=0;
        //如果数组长度为0
        if(nums.length==0)return 0;
        while(fast<nums.length){
            if(nums[slow]!==nums[fast]){
                nums[slow]=nums[fast];
                slow++;
            }
            fast++;
        }
        return slow;
    }
}
// @lc code=end
